# Code for comparing pandas dataframes. You should recognize this! 

def canonicalize_tibble(X):
    var_names = sorted(X.columns)
    Y = X[var_names].copy()
    Y.sort_values(by=var_names, inplace=True)
    Y.reset_index(drop=True, inplace=True)
    return Y

def tibbles_are_equivalent (A, B):
    A_canonical = canonicalize_tibble(A)
    B_canonical = canonicalize_tibble(B)
    cmp = A_canonical.eq(B_canonical)
    return cmp.all().all()

def build_movie_i_matrix(movie_i):
    # Columns used in the output format
    used_cols = ['userId', 'movie_1', 'movie_2', 'rating_adjusted_1', 'rating_adjusted_2']
    
    # Subset userId A rated movie i
    distinct_users = np.unique(adjusted_ratings[adjusted_ratings['movieId'] == movie_i].userId)
    
    movie_i_table = []
    
    # Build the table described above and name it movie_i_table
    # Iterate over the distinct users to see how many other movies they watched
    for user in distinct_users:
        mini_df = pd.DataFrame()
        sub_df = adjusted_ratings[adjusted_ratings['userId'] == user]
        
        if len(sub_df) == 1:    # the user only watched 1 movie
            pass
        else:
            mini_df = sub_df[sub_df['movieId'] != movie_i]
            mini_df['movie_1'] = movie_i
            mini_df['rating_adjusted_1'] = float(sub_df[sub_df['movieId'] == movie_i]['rating_adjusted'])
            mini_df.rename(columns={'movieId':'movie_2',
                                   'rating_adjusted':'rating_adjusted_2'},
                           inplace=True)
            
        movie_i_table.append(mini_df)
    
    ###
    movie_i_table = pd.concat(movie_i_table)
    movie_i_table.reset_index(drop=True, inplace=True)
    
    return movie_i_table[used_cols]

def similarity_score(m_i_matrix, movie_2):
    m1_m2_matrix = m_i_matrix[m_i_matrix['movie_2'] == movie_2]

    ###
    A = m1_m2_matrix['rating_adjusted_1'].tolist()
    B = m1_m2_matrix['rating_adjusted_2'].tolist()
    
    Ai_x_Bi = np.multiply(A, B)
    Sigma_AB = np.sum(Ai_x_Bi)
    Ai_norm = np.sqrt(np.sum([ra1**2 for ra1 in A]))
    Bi_norm = np.sqrt(np.sum([ra2**2 for ra2 in B]))
    
    similar_score = Sigma_AB / (Ai_norm * Bi_norm)
    ###

    return similar_score

# function of building the item-to-item weight matrix
def create_weight_matrix(adjusted_ratings):
    # define weight matrix
    matrix_cols = ['movie_1', 'movie_2', 'weight']
    w_matrix = pd.DataFrame(columns = matrix_cols)

    # list all distinct movies in dataset
    distinct_mv = adjusted_ratings.movieId.unique().tolist()

    # Loop through all movies and apply the functions we built above to build a matrix show similarity score of movies
    # Hint: You might need more than 1 loop ;) 
    
    ###
    for mv in distinct_mv:
        mini_dfs = pd.DataFrame()
        
        mv_i_matrix = build_movie_i_matrix(mv)
        mini_dfs['movie_1'] = mv_i_matrix['movie_1']
        mini_dfs['movie_2'] = mv_i_matrix['movie_2']
        
        weights = []
        for other_mv in mv_i_matrix['movie_2'].tolist():
            weights.append(similarity_score(mv_i_matrix, other_mv))
        mini_dfs['weight']  = weights
        
        w_matrix = pd.concat([w_matrix, mini_dfs])
  
    ###
    w_matrix.drop_duplicates(subset=['movie_1', 'movie_2'], inplace=True)

    return w_matrix

def predict_rating(userId, movieId, w_matrix):

    # rating_mean for each movie
    rating_mean = full_ratings.groupby(['movieId'], 
                                       as_index=False).mean().rename(columns={'rating': 'rating_mean'})[['movieId','rating_mean']]

    # set mean_rating = 2.5 if that movie has not been rated by any user.
    if rating_mean[rating_mean['movieId'] == movieId].shape[0] > 0:
        mean_rating = rating_mean[rating_mean['movieId'] == movieId]['rating_mean'].iloc[0]
    else:
        mean_rating = 2.5

    # Calculate predicted_rating for movieId by userId (based on w_matrix and formula given above)
    # Note: In case the denominator == 0 ( sum of absolute weight_ij == 0 ): 
    # => s(i;u) = Mean_i ~ predicted_rating = mean_rating
    
    ###
    # get df where 
    user_info = adjusted_ratings[adjusted_ratings['userId'] == userId]
    
    numerator_temp = []
    weights = []
    
    for movie, adj_rating in zip(user_info['movieId'], user_info['rating_adjusted']):
        weight = w_matrix[(w_matrix['movie_1']==movieId) & (w_matrix['movie_2'] == movie)]['weight']

        if weight.empty:
            numerator_temp.append(0)
            weights.append(0)
        else:
            numerator_temp.append(float(weight)*adj_rating)
            weights.append(float(weight))
        

    numerator = np.sum(numerator_temp)
    denominator = np.sum(np.abs(weights))

    if denominator == 0:
        denominator = mean_rating
    else:
        denominator
        
    predicted_rating = mean_rating + (numerator/denominator)
    ###
    
    return predicted_rating

def recommend(userId, w_matrix, adjusted_ratings, top_n=10):
    distinct_mv = adjusted_ratings['movieId'].unique()

    user_rating_full = pd.DataFrame(columns=['movieId', 'rating'])

    # Loop through all distinct_mv and calculate the ratings for all movies that the user hasn't rated
    # Store these predicted ratings to user_rating_full dataframe  (only not-yet-rated movies from user "userId")
    # Because we only want to recommend top_n movies with highest predicted ratings, that the user "userId" has not watched/rated.
    # We will show top_n items from your user_rating_full table to user as our recommendation

    ###
    user_seen_movies = adjusted_ratings[adjusted_ratings['userId'] == userId]['movieId'].tolist()
    unseen_movies = movies[~movies['movieId'].isin(user_seen_movies)]['movieId'].tolist()
    
    preds = []
    for unseen in unseen_movies:
        preds.append(predict_rating(userId, unseen, w_matrix))
        
    user_rating_full['movieId'] = unseen_movies
    user_rating_full['rating'] = preds
    
    ###
    # select top 10 movies rated by the user
    recommendations = user_rating_full.sort_values(by=['rating'], ascending=False).head(10)
    recommendations["movieId"] = recommendations["movieId"].astype("int64")
    recommendations = recommendations.merge(movies, on="movieId",how="left")

    return recommendations
